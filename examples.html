Implementing a Playlist

HTML:

<div id="player"></div>
<br/>
<a id="next" href="#">Next</a>

JavaScript:

var playlist = [
  {
    provider: 'youtube',
    id: 'gwvXtfRjonc'
  },
  {
    provider: 'vimeo',
    id: '75423505'
  },
  {
    provider: 'youtube',
    id: 'sxpLiQgdTI0'
  }
];
var opts = {
  autoplay: true
};

var player = $('#player').xmbd();
var index = 0;

var embedNext = function(){
    player.embed(playlist[index].provider, playlist[index].id, opts);
    index += 1;
    if(index === playlist.length){
        index = 0;
    }
};

player.on('vEnded', embedNext);
$("#next").click(embedNext);
embedNext();

Player showcasing available actions

HTML:

<div id="player"></div>
<a class="cmd" href="#">pause</a> |
<a class="cmd" href="#">play</a> |
<a class="cmd" href="#">toggle</a> |
<a class="cmd" href="#">stop</a> |
<a id="cue" href="#">cue</a>
<br>
<a class="mbd" href="#">youtube</a>,
<a class="mbd" href="#">vimeo</a>,
<a class="mbd" href="#">dailymotion</a>

JavaScript:

var vids = {
	youtube: "n_-ayw8sLOA",
	vimeo: "75320274",
	dailymotion: "x1571bg"
};
var opts = {
	autoplay: true,
	autohide: true,
	theme: "light"
};

var x;
$(function(){
	x = $("#player").xmbd();
	
	x.on("playerStateChange", function(e){
		console.log(e);
	});

	x.embed("dailymotion", "x1571bg", opts);
	
	$(".cmd").click(function(e){
		e.preventDefault();
		x.action($(this).text().toLowerCase());
	});
	$(".mbd").click(function(e){
		e.preventDefault();
		var p = $(this).text().toLowerCase();
		x.embed(p, vids[p], opts);
	});
	$("#cue").click(function(e){
		e.preventDefault();
		x.action("cue", 20);
	});
});

JavaScript

var vids = {
	youtube: "n_-ayw8sLOA",
	vimeo: "75320274",
	dailymotion: "x1571bg"
};
var opts = {
	autoplay: true,
	autohide: true,
	theme: "light"
};

var x = $("#player").xmbd();

x.on("playerStateChange", function(e){
    var d = new Date();
    d = d.getHours() + ":" + d.getMinutes() + ":" + (d.getSeconds().toString().length == 1 ? "0" + d.getSeconds() : d.getSeconds()) + " ";
    $("#events").prepend($("
  • ").text(d + e));
});

x.embed("dailymotion", "x1571bg", opts);

$(".cmd").click(function(e){
	e.preventDefault();
	x.action($(this).text().toLowerCase());
});
$(".mbd").click(function(e){
	e.preventDefault();
	var p = $(this).text().toLowerCase();
	x.embed(p, vids[p], opts);
